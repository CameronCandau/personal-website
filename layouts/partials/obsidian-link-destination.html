{{- $page := .page -}} {{- $target := trim .target " " -}} {{- if hasPrefix $target "#" -}} {{- printf "#%s" (anchorize (strings.TrimPrefix "#" $target)) -}} {{- else -}} {{- $candidates := slice -}} {{- if and $page.File (not (hasPrefix $target "/")) (not (strings.Contains $target "://")) -}} {{- $dir := path.Dir $page.File.Path -}} {{- if eq (path.Base $page.File.Path) "index.md" -}} {{- $dir = path.Dir $dir -}} {{- end -}} {{- if ne $dir "." -}} {{- $candidates = $candidates | append (printf "%s/%s" $dir $target) -}} {{- end -}} {{- end -}} {{- $candidates = $candidates | append $target -}} {{- if not (hasPrefix $target "ctf/") -}} {{- $candidates = $candidates | append (printf "ctf/%s" $target) -}} {{- end -}} {{- $resolved := "" -}} {{- range $candidate := $candidates -}} {{- if not $resolved -}} {{- $pageMatch := site.GetPage (replaceRE `\.md$` "" $candidate) -}} {{- if not $pageMatch -}} {{- $pageMatch = site.GetPage (replaceRE `/index$` "" (replaceRE `\.md$` "" $candidate)) -}} {{- end -}} {{- with $pageMatch -}} {{- $resolved = .RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- if $resolved -}} {{- $resolved -}} {{- else -}} {{- $parts := split (replaceRE `(^|/)\.?` "$1" (replaceRE `\.md$` "" $target)) "/" -}} {{- $out := slice -}} {{- range $index, $part := $parts -}} {{- if and $part (ne $part ".") (not (and (eq $index (sub (len $parts) 1)) (eq $part "index"))) -}} {{- $out = $out | append ($part | urlize) -}} {{- end -}} {{- end -}} {{- delimit $out "/" -}} {{- end -}} {{- end -}}
